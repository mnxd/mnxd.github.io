0. delete[]

这个问题其实来自于看到师兄师姐的面经里有提到一个问题

delete[]是如何知道要delete的数组大小的？

Type * pName = new Type[Size];

在执行上述语句时，一开始我简单的以为C++编译器内部实现了某种方法如

  • 申请pName指向的内存空间时会自动在sizeof(Type)*size前面+1块内存用来存储大小

  • 或者通过sizeof(theArray)/Sizeof[Type]得到数组大小（错误想法）

  • 内存管理库自己维护了一个列表,记录所有空间分配的位置和大小以及类型,delete的时候查一下这个列表就行了

1.过分配内存方法

通过查阅资料，似乎在不同的编译器中都有对第一种方法的实现如下，

以下内容来自知乎夏华林的回答 ,这里抄过来做一下笔记。

/* _freea must be in the header so that its allocator matches _malloca */
#if !defined (__midl) && !defined (RC_INVOKED)

再来看看 SGI　STL标准库源码 stl_alloc.h 文件209 行到 246行 debug_alloc类模板的设计：

// Allocator adaptor to check size arguments for debugging.
// Reports errors using assert.  Checking can be disabled with
// NDEBUG, but it's far better to just use the underlying allocator
// instead when no checking is desired.
// There is some evidence that this can confuse Purify.
template 
class debug_alloc {

private:

  enum {_S_extra = 8};  // Size of space used to store size.  Note
                        // that this must be large enough to preserve
                        // alignment.

                        //这儿就像它所说的那样
public:

  static void* allocate(size_t __n)
  {
    // 这里实际申请的内存大小要多 8 个字节
    char* __result = (char*)_Alloc::allocate(__n + (int) _S_extra);
    *(size_t*)__result = __n;//前 4 个字节用于存储区块大小，后4个字节的空白区暂时不清楚是为什么。
    return __result + (int) _S_extra;//最后返回相对于原始指针偏移8个字节的新指针
  }

  static void deallocate(void* __p, size_t __n)
  {
    char* __real_p = (char*)__p - (int) _S_extra;//获得原始指针
    assert(*(size_t*)__real_p == __n);//这里增加了一个断言，防止析构了被破坏的指针
    _Alloc::deallocate(__real_p, __n + (int) _S_extra);
  }

  static void* reallocate(void* __p, size_t __old_sz, size_t __new_sz)
  {
    char* __real_p = (char*)__p - (int) _S_extra;
    assert(*(size_t*)__real_p == __old_sz);
    char* __result = (char*)
      _Alloc::reallocate(__real_p, __old_sz + (int) _S_extra,
                                   __new_sz + (int) _S_extra);
    *(size_t*)__result = __new_sz;
    return __result + (int) _S_extra;
  }

};
